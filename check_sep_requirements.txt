#!/bin/bash
#
# Script de validação para instalação do Symantec Endpoint Protection no Ubuntu 22.04.5
# Verifica versão do kernel, headers e dependências.
#

echo "=== Validação de requisitos para Symantec Endpoint Protection ==="

# 1. Verificar versão do kernel em uso
KERNEL_VERSION=$(uname -r)
KERNEL_MAJOR=$(echo $KERNEL_VERSION | cut -d"." -f1)
KERNEL_MINOR=$(echo $KERNEL_VERSION | cut -d"." -f2)

echo ">> Kernel detectado: $KERNEL_VERSION"

if [ "$KERNEL_MAJOR" -eq 5 ] && [ "$KERNEL_MINOR" -eq 15 ]; then
    echo "[OK] Kernel é compatível (5.15.x)."
else
    echo "[ALERTA] Kernel não é 5.15.x. O SEP pode não ser suportado nesta versão."
fi

# 2. Verificar se headers do kernel estão instalados
dpkg -l | grep -q "linux-headers-$(uname -r)"
if [ $? -eq 0 ]; then
    echo "[OK] Headers do kernel estão instalados."
else
    echo "[FALTA] Headers do kernel não encontrados. Instale com:"
    echo "        sudo apt update && sudo apt install linux-headers-$(uname -r)"
fi

# 3. Verificar se o dmidecode está presente
if command -v dmidecode &> /dev/null; then
    echo "[OK] dmidecode está instalado."
else
    echo "[FALTA] dmidecode não encontrado. Instale com:"
    echo "        sudo apt update && sudo apt install dmidecode"
fi

# 4. Exibir espaço livre (útil para instalação de pacotes)
echo ">> Espaço livre em /:"
df -h / | tail -1 | awk '{print $4 " livres"}'

echo "=== Validação concluída ==="
